  \begin{question}
    $decl \rightarrow TYPE\ list\ ;$\\
    $list \rightarrow IDENT\ list'$\\
    $list' \rightarrow \epsilon$\\
    $list' \rightarrow ,\ IDENT\ list'$\\
    \begin{verbatim}
        decl()
        {
            if next == TYPE
                match(TYPE)
                list()
                match(';')
            else
                error()
        }

        list()
        {
            if next == IDENT
                match(IDENT)
                list'()
            else
                error()
        }

        list'()
        {
            if next == ','
                match(',')
                match(IDENT)
                list'()
            else if next == ';'
                return
            else
                error()
        }
    \end{verbatim}
  \end{question}
