le: nat x nat $\rightarrow$ bool\\
\[\text{le}(n, m) =
  \begin{cases}
    \text{if n is O} & \text{true}\\
    \text{if n is S}n' & \begin{cases}
                          \text{if m is O} & false\\
                          \text{if m is S}m' & le(n',m')
                        \end{cases}
  \end{cases}
\]

\[\text{ge}(n, m) =
  \begin{cases}

                          \text{if n is S}n' & ge(n',m')
                        \end{cases}
  \end{cases}
\]
