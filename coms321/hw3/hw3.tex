  \question{}\vspace{-.5cm}
  \begin{align*}
    \text{new frequencey} &= .4 * .3\\
                   &= .12\\
    \text{CPI}_{\text{new}} &= \frac{(.4 - .12) *2 + (.2-.12)*4 + .18*4 + .22*8 + .12 * 4}{1-.12}\\
                     &= 4.36\\
    \text{CPI}_{\text{old}} &= .4 *2 + .2*4 + .18*4 + .22*6\\
                     &= 3.64\\
    \text{Time}_{\text{old}} &= Ops_{\text{old}} * 3.64 * CCT_{\text{old}}\\
                      &= 3.64 * Ops_{\text{old}} * CCT_{\text{old}}\\
    \text{Time}_{\text{new}} &= Ops_{\text{new}} * 4.36 * CCT_{\text{new}}\\
                      &= 4.36 * (1-.12) * Ops_{\text{old}} * CCT_{\text{old}}\\
                      &= 3.83 * Ops_{\text{old}} * CCT_{\text{old}}
  \end{align*}
  This change would not improve preformance, because the execution time ends up increasing.

  \question{}\vspace{-.5cm}
  \begin{align*}
    \text{CPI} &= .35 *1 + .25*2 + .15*2 + .25*3\\
                     &= 1.9\\
    \text{CPI}_{\text{new}} &= \frac{(.35 - .0875) *1 + (.25-.0875)*2 + .15*2 + .25*5 + .0875 * 1}{1-.0875}\\
                     &= 2.438\\
    \text{Time}_{\text{old}} &= Ops_{\text{old}} * 1.9 * CCT_{\text{old}}\\
                      &= 1.9 * Ops_{\text{old}} * (1.3*CCT_{\text{new}}\\
                      &= 2.47 * Ops_{\text{old}} * CCT_{\text{new}}\\
    \text{Time}_{\text{new}} &= Ops_{\text{new}} * 2.438 * CCT_{\text{new}}\\
                      &= 2.438 * (1-.0875) * Ops_{\text{old}} * CCT_{\text{new}}\\
                      &= 2.225 * Ops_{\text{old}} * CCT_{\text{new}}\\
    \frac{\text{Time}_{\text{old}}}{\text{Time}_{\text{new}}} &= \frac{2.438}{2.225}\\
                                                &= 1.12
  \end{align*}
  the new version is faster by 12\%
